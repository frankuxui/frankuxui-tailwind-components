### My first post


```jsx live=true
() => {
  const [likes, increaseLikes] = React.useState(0);

  return (
    <div>
      <h1>
        {`${likes} likes`}
      </h1>
      <button className='py-2 px-4 bg-red-400' onClick={() => increaseLikes(likes + 1)}>Like</button>
    </div>
  );
}
```
